静默安装oracle

一、前期准备

1、配置解析

​x
vim /etc/hosts
​
192.168.10.164  orcl

 

2、创建Oracle账户以及用户组

xxxxxxxxxx
groupadd -g 1001 dba
​
groupadd -g 1002 oper
​
groupadd -g 1003 oinstall
​
useradd -u 1001 -g oinstall -G dba,oper -d /home/oracle -s /bin/bash oracle
​
echo 'oracle' | passwd --stdin oracle

 

3、建立数据库软件安装的目录结构

x
mkdir -p /u01/app/oracle/product/11.2.0.4/db
​
chown -R oracle:oinstall /u01
​
#chmod -R 775 /u01
​
 
​

​
chown -R oracle:oinstall /software/
​
#将DATABASE安装包上传至/software
​
 
#创建数据文件目录
mkdir -p /oradata/arch
​
chown -R oracle:oinstall /oradata

7、系统限制修改

xxxxxxxxxx
cat >> /etc/security/limits.conf <<EOF
grid             soft     nproc   65536
grid             hard     nproc   65536
grid             soft     nofile  65536
grid             hard     nofile  65536
grid             soft     stack   10240
grid             hard     stack   65536
oracle           soft     nproc   65536
oracle           hard     nproc   65536
oracle           soft     nofile  65536
oracle           hard     nofile  65536
oracle           soft     stack   10240
oracle           hard     stack   65536
oracle           soft     memlock -1
oracle           hard     memlock -1
EOF

 

8、系统环境变量

xxxxxxxxxx
cat >>/etc/profile <<EOF
if [ $USER = "oracle" ]; then
        if [ $SHELL = "/bin/ksh" ]; then
              ulimit -p 16384
              ulimit -n 65536
        else
              ulimit -u 16384 -n 65536
        fi
fi
EOF
​
source /etc/profile

 

9、oracle用户环境变量

xxxxxxxxxx
ora_base='/u01/app/oracle'
ora_home='/u01/app/oracle/product/11.2.0.4/db'
​
cat >>/home/oracle/.bash_profile <<END
export ORACLE_BASE=${ora_base}
export ORACLE_HOME=${ora_home}
export ORACLE_SID=
export ORACLE_TERM=xterm
export ORACLE_OWNER=oracle
export NLS_DATE_FORMATE="YYYY-MM-DD HH24:MI:SS"
export TMP=/tmp
export TNS_ADMIN=\$ORACLE_HOME/network/admin
export ORA_NLS33=\$ORACLE_HOME/ocommon/nls/admin/data
#export NLS_LANG="SIMPLIFIED CHINESE_CHINA.ZHS16GBK" --AL32UTF8 select userenv('LANGUAGE') db_NLS_LANG from dual; 
export NLS_LANG="AMERICAN_CHINA.ZHS16GBK"
export LD_LIBRARY_PATH=\$ORACLE_HOME/lib:\$ORACLE_HOME/network/lib:/lib:/usr/lib:/usr/local/lib
export LIBPATH=\$ORACLE_HOME/lib:\$ORACLE_HOME/network/lib:/lib:/usr/lib:/usr/local/lib
export PATH=\$PATH:/sbin:/usr/lbin:/usr/sbin:\$JAVA_HOME/bin:\$ORACLE_HOME/bin:\$ORACLE_HOME/lib:\$HOME/bin:\$ORACLE_HOME/OPatch:.
stty erase ^H
umask 022
END
​
source /home/oracle/.bash_profile

3、执行静默安装配置文件：

xxxxxxxxxx
#切换root用户添加权限
cd /software/database/response
chmod u+x weizhi_soft.rsp
​
#切回oracle用户
cd /software/database/
sh runInstaller -silent -responseFile /software/database/response/weizhi_soft.rsp
​
#另外开一个终端，查看进程情况：
ps -ef | grep oracle
#看到一大篇子乱码为正常
​
#最后提示 运行两个脚本（使用root用户）
#As a root user, execute the following script(s):
#/u01/app/oraInventory/orainstRoot.sh
#/u01/app/oracle/product/11.2.0.4/db/root.sh
#Successfully Setup Software.

 

4、成功

3、开始手动建库

xxxxxxxxxx
