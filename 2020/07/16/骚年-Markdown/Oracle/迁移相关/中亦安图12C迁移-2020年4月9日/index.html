1.8 清理迁移用户的连接会话

统计迁移用户的连接信息：

select inst_id,username,program,status,count(*) from gv$session where username in (‘JZTA_HIS’) group by inst_id,username,program,status order by 1,2,3,4;

将迁移用户的连接结束：

select s.username,sid,spid,s.program,s.LAST_CALL_ET s from v$session s,v$process p where p.addr=s.paddr and s.username in (‘JZTA_HIS’);

select ‘ho kill -9 ‘||spid from v$session s,v$process p where p.addr=s.paddr and s.username in (‘JZTA_HIS’);

再次统计迁移用户的连接信息，返回空则继续：

select inst_id,username,program,status,count(*) from gv$session where username in (‘JZTA_HIS’) group by inst_id,username,program,status order by 1,2,3,4;

注：原库其他用户连接不受影响

​ COLUMN free_size format 999,999 heading “Free(MB)”

SELECT a.tablespace_name ts_name

,a.unalloc_size

,nvl(f.free_size, 0) free_size

,a.used_size - nvl(f.free_size, 0) used_size

,round((a.used_size - nvl(f.free_size, 0)) /

​ (a.unalloc_size + a.used_size)

​ ,2)*100 capacity

FROM (SELECT tablespace_name

​ ,round(SUM(bytes) / 1024 / 1024) free_size

​ ,round(AVG(bytes)/1024/1024) avg_size

​ FROM dba_free_space

​ GROUP BY tablespace_name) f

,(SELECT tablespace_name

​ ,round(SUM(user_bytes) / 1024 / 1024) used_size

​ ,round(SUM(decode(autoextensible

​ ,’YES’

​ ,decode(sign(maxbytes - user_bytes)

​ ,-1

​ ,0

​ ,maxbytes - user_bytes)

​ ,0)) / 1024 / 1024) unalloc_size

​ FROM dba_data_files

​ WHERE status=’AVAILABLE’

​ GROUP BY tablespace_name) a

WHERE 1 = 1

AND a.tablespace_name = f.tablespace_name(+)

ORDER BY capacity DESC;
